42grille_t creation_grille(int bord){
43 grille_t g;
44 int i,j;
45 int taille=64;
46 for(i=0;i<LONG;i++){
47 for(j=0;j<LARG;j++){
48 g.tabGrille[i][j]=creation_case();
49 switch (bord){
50 case 0 : if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;break;
51 case 1 : if(j == 0 )g.tabGrille[i][j].etat = 3;
52 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
53 break; //N
54 case 2 : if(i == (LONG-1) )g.tabGrille[i][j].etat = 3;
55 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
56 break; //E
57 case 3 : if(j == (LARG-1) )g.tabGrille[i][j].etat = 3;
58 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
59 break; //S
60 case 4 : if(i == 0 )g.tabGrille[i][j].etat = 3;
61 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
62 break; //O
63 case 5 : if(j == 0 || i == (LONG-1) )g.tabGrille[i][j].etat = 3;
64 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
65 break; //NE
66 case 6 : if(j == (LARG-1) || i == (LONG-1) )g.tabGrille[i][j].etat = 3;
67 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
68 break; //SE
69 case 7 : if(j == (LARG-1) || i == 0 )g.tabGrille[i][j].etat = 3;
70 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
71 break; //SO
72 case 8 : if(j == 0 || i == 0 )g.tabGrille[i][j].etat = 3;
73 else if(i == 0 || i == (LONG-1) || j == 0 || j == (LARG-1))g.tabGrille[i][j].etat = 2;
74 break; //NO
75 default: break;
76 }
77 g.tabGrille[i][j].Rectangle.x=taille*i;
78 g.tabGrille[i][j].Rectangle.y=taille*j+56;
79 g.tabGrille[i][j].x=i;
80 g.tabGrille[i][j].y=j;
81 }
82 }
83 return g;

1114int afficher_map(SDL_Event event,map_t map, SDL_Renderer *renderer, int *we, int *he, int *etat_map,carte_t * cartec){
1115 int i, j;
1116
1117 SDL_Rect Rectangle;
1118 SDL_Rect Rectanglep;
1119
1120 // Marges
1121 float margeX = (*we) * 0.1;
1122 float margeY = (*he) * 0.1;
1123
1124 // Largeur et hauteur utilisables
1125 float largeurUtilisable = (*we) - 2 * margeX;
1126 float hauteurUtilisable = (*he) - 2 * margeY;
1127
1128 // Taille de chaque carré
1129 float taille_carre = min(largeurUtilisable, hauteurUtilisable) / 6;
1130
1131 //Taille du petit carré de l'emplacement actuel
1132 float taille_pcarre = taille_carre/3;
1133
1134 // Coordonnées du premier carré
1135 float firstX = margeX + ((*we) - 2 * margeX - taille_carre * 6) / 2;
1136 float firstY = margeY + ((*he) - 2 * margeY - taille_carre * 6) / 2;
1137 Rectangle.x = (int)firstX;
1138 Rectangle.y = (int)firstY;
1139 Rectangle.w = (int)taille_carre;
1140 Rectangle.h = (int)taille_carre;
1141
1142 //Initialisation du petit carré de l'emplacement actuel
1143 Rectanglep.x = (int)firstX + (taille_pcarre);
1144 Rectanglep.y = (int)firstY + (taille_pcarre);
1145 Rectanglep.w=(int)taille_pcarre;
1146 Rectanglep.h=(int)taille_pcarre;
1147
1148 if((*etat_map) == 1){
1149
1150 SDL_Surface* surface = SDL_CreateRGBSurface(0, *we, *he, 32,0xFF000000, 0x00FF0000, 0x0000FF00, 0x000000FF);
1151 // Copie les pixels du rendu actuel dans la surface
1152 SDL_RenderReadPixels(renderer, NULL, surface->format->format, surface->pixels, surface->pitch);
1153 // Créer une texture à partir de la surface
1154 SDL_Texture* targetTexture = SDL_CreateTextureFromSurface(renderer, surface);
1155 SDL_FreeSurface(surface);
1156
1157 SDL_Rect tecran = {0,0,*we,*he};
1158
1159 while((*etat_map) < 2){
1160 while (SDL_PollEvent(&event) != 0) {
1161 if(event.type == SDL_KEYDOWN && (event.key.keysym.sym == SDLK_x)){
1162 (*etat_map)=2;
1163 }
1164 }
1165 SDL_RenderClear(renderer);
1166 SDL_RenderCopy(renderer, targetTexture, NULL, NULL);
1167 SDL_SetRenderDrawColor(renderer,0,0,0,100);
1168 SDL_RenderFillRect(renderer, &tecran);
1169 for(i=0;i<ROWS;i++){
1170 Rectangle.y = (int)(firstY + i * taille_carre);
1171 for(j=0;j<COLUMNS;j++){
1172 if(map.tabMap[i][j].etat_brouillard != 1){
1173 switch(map.tabMap[i][j].nZone){
1174 case 1: SDL_SetRenderDrawColor(renderer, 206,206,206,255);break;
1175 case 2: SDL_SetRenderDrawColor(renderer, 86,115,154,255);break;
1176 case 3: SDL_SetRenderDrawColor(renderer, 153,81,43,255);break;
1177 case 4: SDL_SetRenderDrawColor(renderer, 104,157,113,255);break;
1178 case 5: SDL_SetRenderDrawColor(renderer, 115,8,0,255);break;
1179 default: break;
1180 }
1181 }else{
1182 SDL_SetRenderDrawColor(renderer, 0,0,0,255);
1183 }
1184 Rectangle.x = (int)(firstX + j * taille_carre);
1185 SDL_RenderFillRect(renderer, &Rectangle);
1186 if(cartec->ycarte == j && cartec->xcarte == i){
1187 SDL_SetRenderDrawColor(renderer, 255,255,0,255);
1188 Rectanglep.x = Rectangle.x+taille_pcarre;
1189 Rectanglep.y = Rectangle.y+taille_pcarre;;
1190 SDL_RenderFillRect(renderer, &Rectanglep);
1191 }
1192 }
1193 }
1194 SDL_RenderPresent(renderer);
1195 SDL_Delay(100);
1196 }
1197 (*etat_map) = 0;

1199 SDL_RenderCopy(renderer, targetTexture, NULL, NULL);
1200 SDL_RenderPresent(renderer);
